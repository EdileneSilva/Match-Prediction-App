Pour build le container FastAPI_App :
    
    cd /Users/slyxi7/Desktop/Match-Prediction-App/FastAPI_App
    docker build -f Dockerfile.api -t fastapi_api .

Pour build le container FastAPI_ML :
    
    cd /Users/slyxi7/Desktop/Match-Prediction-App/FastAPI_ML
    docker build -f Dockerfile -t fastapi_ml .

Pour build le container Frontend :
    
    cd /Users/slyxi7/Desktop/Match-Prediction-App/match_prediction_app-front
    docker build -f Dockerfile -t frontend_app .

Pour supprimer des containers spécifiques : 
    
    docker rm -f fastapi_api fastapi_ml frontend_app postgres_db

Pour supprimer les containers et leurs images : 

    # Supprimer les containers
    docker rm -f $(docker ps -aq)

    # Supprimer les images associées
    docker rmi fastapi_api fastapi_ml frontend_app

Pour start tout les containers d'un coup :

    docker start postgres_db fastapi_api fastapi_ml frontend_app

Pour arreter tout les containers d'un coup :

    docker stop postgres_db fastapi_api fastapi_ml frontend_app
